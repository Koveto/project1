# battle/battle_renderer.py

import pygame
from pokedex.pokemon_sprites import load_pokemon_sprite, load_player_sprite

class BattleRenderer:

    def __init__(self, background_surface, model):
        """
        background_surface: loaded battle background
        model: BattleModel instance
        """
        self.background = background_surface
        self.model = model

        self.player_sprite = load_player_sprite(
            scale=4
        )

        self.mewtwo_sprite = load_pokemon_sprite(
            gen=1,
            index=((150*2)-2),
            is_shiny=False,
            is_back=False,
            scale=4
        )

        self.blastoise_sprite = load_pokemon_sprite(
            gen=1,
            index=((9*2)-2),
            is_shiny=False,
            is_back=False,
            scale=4
        )

        self.venusaur_sprite = load_pokemon_sprite(
            gen=1,
            index=((3*2)-2),
            is_shiny=False,
            is_back=False,
            scale=4
        )

        self.butterfree_sprite = load_pokemon_sprite(
            gen=1,
            index=((12*2)-2),
            is_shiny=False,
            is_back=False,
            scale=4
        )

        self.nidoking_sprite = load_pokemon_sprite(
            gen=1,
            index=((34)-1),
            is_shiny=False,
            is_back=True,
            scale=4
        )

        self.kangaskhan_sprite = load_pokemon_sprite(
            gen=1,
            index=((115)-1),
            is_shiny=False,
            is_back=True,
            scale=4
        )

        self.charizard_sprite = load_pokemon_sprite(
            gen=1,
            index=((6)-1),
            is_shiny=False,
            is_back=True,
            scale=4
        )

    def draw(self, screen):
        # Background
        screen.blit(self.background, (0, 0))

        DIST_BETWEEN = 150
        X_FIRST = -64
        PLAYER_OFFSET = 50
        X_FIRST_ENEMY = 148
        DIST_BETWEEN_ENEMY = 182

        # Pok√©mon sprites
        screen.blit(self.butterfree_sprite, (X_FIRST_ENEMY + (3*DIST_BETWEEN_ENEMY), 0))
        screen.blit(self.venusaur_sprite, (X_FIRST_ENEMY + (1*DIST_BETWEEN_ENEMY), 0))
        screen.blit(self.mewtwo_sprite, (X_FIRST_ENEMY + (2*DIST_BETWEEN_ENEMY), 0))
        screen.blit(self.blastoise_sprite, (X_FIRST_ENEMY + (0*DIST_BETWEEN_ENEMY), 0))
        screen.blit(self.nidoking_sprite, (X_FIRST, 192))
        screen.blit(self.player_sprite, (PLAYER_OFFSET + X_FIRST + (1*DIST_BETWEEN), 240))
        screen.blit(self.kangaskhan_sprite, ((X_FIRST + (2*DIST_BETWEEN)), 192))
        screen.blit(self.charizard_sprite, ((X_FIRST + (3*DIST_BETWEEN)), 192))

        # Future: HP bars, UI menus, text boxes
